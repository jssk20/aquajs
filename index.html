Quick Setup Steps
1. Flash the code (right panel) via Arduino IDE at 115200 baud
2. Close Arduino Serial Monitor completely
3. Open this dashboard in Chrome or Edge (not Firefox)
4. Click "Connect ESP32" → select your COM port
5. Real TDS values appear instantly on Dashboard
ESP32 Arduino Code
// AquaDrone ESP32 — TDS Sensor + Relay Pump // TDS AOUT → GPIO 34 (ADC, input-only pin) // Relay IN1 → GPIO 26 (digital output, active-LOW) #define TDS_PIN 34 #define RELAY_PIN 26 #define VREF 3.3 #define ADC_RES 4096.0 float kValue = 1.0; // calibration factor bool pumpOn = false; bool autoMode = true; int cycles = 0; unsigned long lastRead = 0; unsigned long pumpStart= 0, totalMs = 0; void setup() { Serial.begin(115200); analogReadResolution(12); pinMode(RELAY_PIN, OUTPUT); digitalWrite(RELAY_PIN, HIGH); // HIGH = relay OFF Serial.println("READY"); } float readTDS() { float sum = 0; for(int i=0; i<30; i++){ sum += analogRead(TDS_PIN); delay(10); } float v = (sum/30.0) * VREF / ADC_RES; return constrain( (133.42*v*v*v - 255.86*v*v + 857.39*v)*0.5*kValue, 0, 1500); } void setPump(bool on) { if(on == pumpOn) return; pumpOn = on; digitalWrite(RELAY_PIN, on ? LOW : HIGH); if(on){ pumpStart = millis(); cycles++; } else { totalMs += millis() - pumpStart; } Serial.println(on ? "PUMP:ON" : "PUMP:OFF"); } void loop() { if(millis() - lastRead >= 1500) { lastRead = millis(); float tds = readTDS(); unsigned long runMs = pumpOn ? totalMs + (millis()-pumpStart) : totalMs; // CSV line dashboard parses: // TDS:342.5,PUMP:OFF,CYC:2,RUN:45000,UP:90000 Serial.print("TDS:"); Serial.print(tds, 1); Serial.print(",PUMP:"); Serial.print(pumpOn?"ON":"OFF"); Serial.print(",CYC:"); Serial.print(cycles); Serial.print(",RUN:"); Serial.print(runMs); Serial.print(",UP:"); Serial.println(millis()); if(autoMode) { if(!pumpOn && tds > 500) setPump(true); if( pumpOn && tds < 200) setPump(false); } } // Commands sent from dashboard via serial: // ON / OFF / AUTO / MANUAL / K=1.2 if(Serial.available()) { String c = Serial.readStringUntil('\n'); c.trim(); c.toUpperCase(); if (c == "ON") { autoMode=false; setPump(true); } else if(c == "OFF") { autoMode=false; setPump(false); } else if(c == "AUTO") autoMode = true; else if(c == "MANUAL") autoMode = false; else if(c.startsWith("K=")) kValue = c.substring(2).toFloat(); } }
