⚠ HC-SR04 ECHO outputs 5V — use a voltage divider (1kΩ + 2kΩ) to drop it to 3.3V before GPIO 14.
⚠ GPIO 34 is input-only. Relay IN1 is active-LOW (LOW = pump ON).
Setup Steps
1. Flash the code (right panel) at 115200 baud
2. Mount ultrasonic sensor pointing down into bottle neck
3. Set BOTTLE_DEPTH_CM to your actual bottle height in the code
4. Close Arduino Serial Monitor — open this dashboard in Chrome/Edge
5. Click Connect ESP32 → select COM port → data streams live
ESP32 Arduino Code
// AquaDrone ESP32 // TDS → GPIO 34 (ADC, input-only) // Relay → GPIO 26 (active-LOW) // TRIG → GPIO 27 // ECHO → GPIO 14 (use voltage divider! 5V→3.3V) #define TDS_PIN 34 #define RELAY_PIN 26 #define TRIG_PIN 27 #define ECHO_PIN 14 #define VREF 3.3 #define ADC_RES 4096.0 #define BOTTLE_DEPTH_CM 20.0 // your bottle height in cm #define STOP_FILL_PCT 90 // stop pump at 90% full #define TDS_THRESHOLD 100 // >= 100 ppm = abnormal float kValue = 1.0; bool pumpOn = false; bool autoMode = true; int cycles = 0; unsigned long lastRead = 0; unsigned long pumpStart = 0, totalMs = 0; void setup() { Serial.begin(115200); analogReadResolution(12); pinMode(RELAY_PIN, OUTPUT); pinMode(TRIG_PIN, OUTPUT); pinMode(ECHO_PIN, INPUT); digitalWrite(RELAY_PIN, HIGH); // OFF at start Serial.println("READY"); } // ── TDS Reading ───────────────────────────────────────── float readTDS() { float sum = 0; for(int i=0; i<30; i++){ sum += analogRead(TDS_PIN); delay(10); } float v = (sum/30.0) * VREF / ADC_RES; return constrain( (133.42*v*v*v - 255.86*v*v + 857.39*v)*0.5*kValue, 0, 1500); } // ── Ultrasonic Distance ────────────────────────────────── float readDistance() { digitalWrite(TRIG_PIN, LOW); delayMicroseconds(2); digitalWrite(TRIG_PIN, HIGH); delayMicroseconds(10); digitalWrite(TRIG_PIN, LOW); long dur = pulseIn(ECHO_PIN, HIGH, 30000); // 30ms timeout if(dur == 0) return -1; // no echo return dur * 0.0343 / 2.0; // cm } // ── Fill % from distance ───────────────────────────────── int getFillPct(float distCm) { if(distCm < 0) return -1; // sensor at top: small distance = nearly full float filled = BOTTLE_DEPTH_CM - distCm; int pct = (int)((filled / BOTTLE_DEPTH_CM) * 100.0); return constrain(pct, 0, 100); } // ── Pump Control ───────────────────────────────────────── void setPump(bool on) { if(on == pumpOn) return; pumpOn = on; digitalWrite(RELAY_PIN, on ? LOW : HIGH); if(on){ pumpStart = millis(); cycles++; } else { totalMs += millis() - pumpStart; } Serial.println(on ? "PUMP:ON" : "PUMP:OFF"); } // ── Main Loop ──────────────────────────────────────────── void loop() { if(millis() - lastRead >= 1500) { lastRead = millis(); float tds = readTDS(); float distCm = readDistance(); int fillPct = getFillPct(distCm); unsigned long runMs = pumpOn ? totalMs + (millis()-pumpStart) : totalMs; // Serial output parsed by dashboard: // TDS:85.2,PUMP:ON,CYC:1,RUN:5000,UP:12000,DIST:8.4,FILL:58 Serial.print("TDS:"); Serial.print(tds,1); Serial.print(",PUMP:"); Serial.print(pumpOn?"ON":"OFF"); Serial.print(",CYC:"); Serial.print(cycles); Serial.print(",RUN:"); Serial.print(runMs); Serial.print(",UP:"); Serial.print(millis()); Serial.print(",DIST:"); Serial.print(distCm,1); Serial.print(",FILL:"); Serial.println(fillPct); // Auto-stop pump when bottle reaches 90% full if(pumpOn && fillPct >= STOP_FILL_PCT) { setPump(false); Serial.println("BOTTLE_FULL"); } } // Commands from dashboard: ON / OFF / AUTO / MANUAL / K=x if(Serial.available()) { String c = Serial.readStringUntil('\n'); c.trim(); c.toUpperCase(); if (c=="ON") { autoMode=false; setPump(true); } else if(c=="OFF") { autoMode=false; setPump(false); } else if(c=="AUTO") autoMode=true; else if(c=="MANUAL") autoMode=false; else if(c.startsWith("K=")) kValue = c.substring(2).toFloat(); } }
